#include <Arduino.h>
#include <Wire.h>

#define SDA_PIN 21
#define SCL_PIN 22

void setup() {
    Serial.begin(115200);
    delay(1000);
    Wire.begin(SDA_PIN, SCL_PIN);
    Serial.println("Escaneando I2C...");

    int found = 0;
    for (uint8_t addr = 1; addr < 127; addr++) {
        Wire.beginTransmission(addr);
        if (Wire.endTransmission() == 0) {
            Serial.printf("Dispositivo encontrado: 0x%02X\n", addr);
            found++;
        }
    }
    if (found == 0) Serial.println("Nenhum dispositivo encontrado — verifique a fiacao");
}

void loop() {}
